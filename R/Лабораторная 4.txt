Индивидульная:

# Определяем функцию
f <- function(x) {
  sqrt(5 - 4*x - x^2)
}

# Генерируем последовательность значений x в области определения [-5, 1]
x_values <- seq(-5, 1, by = 0.001)

# Вычисляем значения функции для этих x
y_values <- sapply(x_values, f)

# Находим максимум функции и соответствующее значение x
max_y <- max(y_values, na.rm = TRUE)
max_x <- x_values[which.max(y_values)]

cat("Наибольшее значение функции y = sqrt(5 - 4x - x^2) равно", max_y, "и достигается при x =", max_x, "\n")
