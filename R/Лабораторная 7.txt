f <- expression(sqrt(x))

# Вычисление производной функции
f_prime <- D(f, "x")

# Подстановка значения x = 4
result <- eval(f_prime, list(x = 4))

# Вывод результата

# Создаем функции для суммарных показателей по дням (продажи и посещаемость)
sales_function <- function(x) {
  return(0.5 * x^2 + 10)  # Пример квадратичной функции для продаж
}

visits_function <- function(x) {
  return(0.3 * x^2 + 50)  # Пример квадратичной функции для посещаемости
}

# Построение графиков с использованием функции curve()
curve(sales_function, from = 1, to = 30, xlab = "Дни", ylab = "Продажи", main = "Суммарные продажи по дням")
curve(visits_function, from = 1, to = 30, xlab = "Дни", ylab = "Посещаемость", main = "Суммарная посещаемость по дням", col = "red")
