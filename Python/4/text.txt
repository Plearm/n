def count_word_occurrences(filename, word1, word2):
    with open(filename, 'r', encoding='utf-8') as file:
        text = file.read().lower()  # Преобразуем текст в нижний регистр для учёта регистра слов

    occurrences_word1 = text.count(word1.lower())
    occurrences_word2 = text.count(word2.lower())

    # Разбиваем текст на слова и проверяем совместные вхождения word1 и word2
    words = text.split()
    occurrences_together = sum(1 for i in range(len(words) - 1) if words[i] == word1.lower() and words[i + 1] == word2.lower())

    return occurrences_word1, occurrences_word2, occurrences_together

filename = "text.txt"
word1 = "слово1"
word2 = "слово2"

result = count_word_occurrences(filename, word1, word2)
print(f"Количество вхождений {word1}: {result[0]}")
print(f"Количество вхождений {word2}: {result[1]}")
print(f"Количество совместных вхождений {word1} и {word2}: {result[2]}")



Это пример текста, в котором встречаются слова слово1 и слово2. Слово1 может встречаться несколько раз в этом тексте, но слово2 встречается только один раз. Слово1 и слово2 могут также встречаться рядом друг с другом, например, слово1 слово2 слово1.

Количество вхождений слово1: 3
